5. JavaScript Initialization

How to activate Living Ink in your page:

// Import the module
import LivingInk from '/assets/js/visions/living-ink.js';

// Initialize on page load
document.addEventListener('DOMContentLoaded', () => {
  LivingInk.init();
});

// Or with VisionController
import { VisionController } from '/assets/js/vision-controller.js';

VisionController.register('living-ink', LivingInk);
VisionController.activate('living-ink');

API Methods

// Initialize
LivingInk.init(options)

// Cleanup
LivingInk.destroy()

// Advanced: Create custom diagrams
const diagram = LivingInk.createSVGDiagram('bagua', {
  strokeColor: '#000',
  strokeWidth: 2,
  duration: 3
});

// Access path library
console.log(LivingInk.paths.bagua);
