**GSAP Animation (LOCKED decisions — sequence, timing from Research):**

Use `onMount` with dynamic imports. Use `gsap.matchMedia()` for reduced motion.

```typescript
import { onMount, onDestroy } from 'svelte';
import { browser } from '$app/environment';

let heroSection: HTMLElement;
let matchMedia: gsap.MatchMedia | null = null;

onMount(async () => {
  if (!browser) return;

  const { gsap } = await import('gsap');
  const { DrawSVGPlugin } = await import('gsap/DrawSVGPlugin');
  const { SplitText } = await import('gsap/SplitText');
  const { ScrollTrigger } = await import('gsap/ScrollTrigger');

  gsap.registerPlugin(DrawSVGPlugin, SplitText, ScrollTrigger);

  matchMedia = gsap.matchMedia();

  // BRANCH: Animate (no motion preference)
  matchMedia.add("(prefers-reduced-motion: no-preference)", () => {
    const tl = gsap.timeline();

    // Phase 1: SVG paths draw in (0 to ~0.6s)
    tl.from('.hero-path', {
      drawSVG: 0,
      duration: 0.6,
      stagger: 0.03,
      ease: 'power3.out'
    });

    // Phase 2: Headline reveal (0.4s to 0.8s, overlaps SVG by 0.2s)
    tl.from('.hero-headline', {
      opacity: 0,
      y: 30,
      duration: 0.5,
      ease: 'power3.out'
    }, '-=0.2');

    // Phase 3: Tagline (0.5s to 0.85s)
    tl.from('.hero-tagline', {
      opacity: 0,
      y: 20,
      duration: 0.35,
      ease: 'power3.out'
    }, '-=0.15');

    // Phase 4: Subline (overlaps with tagline)
    tl.from('.hero-subline', {
      opacity: 0,
      y: 15,
      duration: 0.3,
      ease: 'power3.out'
    }, '-=0.1');

    // Phase 5: CTAs fade in (0.7s to 1.0s)
    tl.from('.hero-ctas', {
      opacity: 0,
      duration: 0.3,
      ease: 'power3.out'
    }, '-=0.1');

    // Phase 6: Carousel fades in (last)
    tl.from('.hero-carousel', {
      opacity: 0,
      duration: 0.4,
      ease: 'power3.out'
    }, '-=0.2');

    // PARALLAX: Illustration moves slower than scroll (basic, not deferred depth layers)
    gsap.to('.hero-bg', {
      yPercent: -12,
      ease: 'none',
      scrollTrigger: {
        trigger: heroSection,
        start: 'top top',
        end: 'bottom top',
        scrub: 1
      }
    });

    return () => {
      tl.kill();
    };
  });

  // BRANCH: Reduced motion — all visible immediately
  matchMedia.add("(prefers-reduced-motion: reduce)", () => {
    gsap.set('.hero-path', { drawSVG: '0% 100%' });
    gsap.set(['.hero-headline', '.hero-tagline', '.hero-subline', '.hero-ctas', '.hero-carousel'], {
      opacity: 1,
      y: 0
    });
  });
});

onDestroy(() => {
  if (matchMedia) {
    matchMedia.revert();
    matchMedia = null;
  }
});
```

**Bind reference:**
Use `bind:this={heroSection}` on the `<section>` element for ScrollTrigger targeting.

**Background color:**
Use `background-color: var(--color-surface)` (cream) matching the existing design system.

**Important anti-patterns to avoid (from Research):**
- Do NOT use `$effect` for animation initialization — use `onMount`
- Do NOT use `setTimeout` for sequencing — use GSAP timeline position parameters
- Do NOT set initial GSAP states via `gsap.set()` before the timeline — use CSS `opacity: 0` on child components
- Do NOT forget `onDestroy` cleanup — `matchMedia.revert()` handles all contexts
- Use `drawSVG: '0% 102%'` as the "to" state if Firefox rendering issues appear (see Pitfall 2) — but start with standard values
